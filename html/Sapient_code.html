def data_sampling(X, y, method = 'smote', kind = 'regular', seed = 0):
    if method == 'smote':
        sm = SMOTE(kind = kind)
        X_res, y_res = sm.fit_sample(X, y)
    
    return {'X' : X_res, 'y' : y_res}
